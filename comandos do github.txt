do git bash---

git add - mandar os arquivos para o repositório

git add (.) -  manda tudo

-m - mensagem

git commit - colocar na lista de push os arquivos

git status - ver os arquivos que estão modificados ou faltando

git pull - fazer download nas modificações dos arquivos ou arquivos que estão faltando

git switch - troca de branchs

git restore (arquivo) - restaura/baixar o arquivo de volta 

-pra mandar arquivos

1. git add (arquivo) 
2. git commit -m "(mensagem)
3. git push - pra dar upload

-caso erro

1.git status
2.git pull
